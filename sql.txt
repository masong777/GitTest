select PRODUCT_NAME,SUM(DETAIL_TOTAL_PRICE) as SUMER from order_detail 
left join product on order_detail.DETAIL_PRODUCT_ID =product.PRODUCT_ID
WHERE DETAIL_DATE_TIME between'2019-01-01 00:00:00' and '2019-12-31 23:59:59'
GROUP BY DETAIL_PRODUCT_ID 
ORDER BY SUMER DESC  limit 0,8
